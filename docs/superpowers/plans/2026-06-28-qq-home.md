function homeUsesQQLogic() {
  return homePreferredProvider() === 'qq';
}
function qqHomePlaylists() {
  return (userPlaylists || []).filter(function(pl){
    return pl && normalizeProviderKey(pl.provider || pl.source) === 'qq';
  });
}
function qqFavoritePlaylist() {
  var lists = qqHomePlaylists();
  return lists.find(function(pl){
    return Number(pl.specialType || 0) === 5 || /我喜欢|喜欢的音乐|liked/i.test(String(pl.name || ''));
  }) || lists.find(function(pl){ return !!pl.subscribed; }) || lists[0] || null;
}
function qqRecentRecord(summary) {
  summary = summary || homeListenSummary();
  var recent = (listenRecords || []).find(function(record){
    return record && listenRecordProvider(record) === 'qq';
  });
  if (recent) return recent;
  if (summary && summary.recent && songProviderKey(summary.recent) === 'qq') return summary.recent;
  return null;
}
function qqSearchSeed(summary) {
  summary = summary || homeListenSummary();
  var recent = qqRecentRecord(summary);
  if (recent && (recent.artist || recent.name)) return recent.artist || recent.name;
  if (summary && summary.topArtist && summary.topArtist.name) return summary.topArtist.name;
  return 'QQ 音乐 推荐';
}
function qqAccountStatusText() {
  if (!(qqLoginStatus && qqLoginStatus.loggedIn)) return '登录后同步 QQ 歌单';
  if (qqLoginStatus.stale) return 'QQ 登录状态可能已失效';
  if (!qqLoginStatus.playbackKeyReady) return 'QQ 已登录 · 播放授权不完整';
  return (qqLoginStatus.nickname || 'QQ 音乐') + ' · 播放授权正常';
}
```

function qqHomeTiles(summary) {
  summary = summary || homeListenSummary();
  var lists = qqHomePlaylists();
  var recent = qqRecentRecord(summary);
  var seed = qqSearchSeed(summary);
  var tiles = [];
  if (lists[0]) tiles.push({ kind: 'qqPlaylist', playlist: lists[0], title: lists[0].name || 'QQ 歌单', sub: ((lists[0].trackCount || 0) ? lists[0].trackCount + ' 首' : 'QQ 歌单'), cover: lists[0].cover });
  if (lists[1]) tiles.push({ kind: 'qqPlaylist', playlist: lists[1], title: lists[1].name || 'QQ 歌单', sub: ((lists[1].trackCount || 0) ? lists[1].trackCount + ' 首' : 'QQ 收藏'), cover: lists[1].cover });
  if (recent) tiles.push({ kind: 'qqRecentSong', song: recent, title: recent.name || '最近 QQ 歌曲', sub: recent.artist || 'QQ 音乐', cover: recent.cover });
  if (recent && recent.artist) tiles.push({ kind: 'qqArtist', query: recent.artist, song: recent, title: recent.artist, sub: 'QQ 歌手入口', cover: recent.cover });
  tiles.push({ kind: 'qqSearch', query: seed, title: 'QQ 搜索开播', sub: seed, cover: recent && recent.cover || (lists[0] && lists[0].cover) || '' });
  while (tiles.length < 5) {
    tiles.push({ kind: 'qqSearch', query: 'QQ 音乐 推荐', title: 'QQ 推荐', sub: '打开 QQ 搜索', cover: '' });
  }
  return tiles.slice(0, 5);
}
```
